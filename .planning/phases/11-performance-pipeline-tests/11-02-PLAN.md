```python
    m.default_confidence = 0.6
    m.source = ClassificationSource.DEFAULT
    m.classify.return_value = result

    """Helper: inject classifiers into pipeline bypassing __init__."""
    pipeline = ClassificationPipeline.__new__(ClassificationPipeline)

    pipeline._init_lock = threading.Lock()

    pipeline._circuit_breaker = None
    return pipeline
```

```python
# Pipeline iterates self._classifiers in order.
# On classifier.classify() returning non-None result: records winner, breaks loop.

# On classifier._enabled being False: the pipeline itself does NOT check enabled —
#   the classifier's own classify() method returns None when self._enabled is False.
```
